fun isPrime of n is 
    var a <- True; 
    var i <- 2; 
    while (i<n and a == True) do 
        if (n%i == 0) then 
            a <- False; 
        endif; 
        i <- i + 1; 
        print a; 
    endwhile; 
    returns a
endfun;
